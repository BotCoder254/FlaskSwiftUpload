Core API Reference

SwiftUpload

from flask_swift_upload import SwiftUpload

app = Flask(__name__)
swift_upload = SwiftUpload(app)

Parameters

Parameter Type Description
app Flask Flask application instance
config dict Optional configuration dictionary

Methods

handle_upload

@app.route('/upload', methods=['POST'])
def upload():
    return swift_upload.handle_upload(
        allowed_extensions=['jpg', 'png'],
        max_size=10 * 1024 * 1024  # 10MB
    )

Parameters

Parameter Type Default Description
allowed_extensions list None List of allowed file extensions
max_size int None Maximum file size in bytes

Returns

{
    'success': bool,
    'file_id': str,
    'message': str,
    'error': str (optional)
}

handle_chunked_upload

@app.route('/upload/chunked', methods=['POST'])
def chunked_upload():
    return swift_upload.handle_chunked_upload(
        chunk_size=1024 * 1024,  # 1MB chunks
        temp_dir='uploads/temp'
    )

Parameters

Parameter Type Default Description
chunk_size int 1048576 Size of each chunk in bytes
temp_dir str 'tmp' Temporary directory for chunks

Returns

{
    'success': bool,
    'chunk_id': str,
    'progress': float,
    'complete': bool,
    'file_id': str (if complete),
    'error': str (optional)
}
